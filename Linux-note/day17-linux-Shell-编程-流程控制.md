#### 2.case语句

```
- 基本语法
case $变量名 in
"值1")
代码1
;;
"值2")
代码2
;;
...省略其他分支...
*)
代码*
;;
esac

```

- 应用实例
案例1:当命令行参数是1时,输出"周一",是2时，就输出"周二"，其他情况输出"other"

```
#!/bin/bash
case $A in
"1")
echo "周一"
;;
"1")
echo "周二"
;;
*)
echo "other..."
;;
esac
```

#### 3.for循环

```
- 基本语法1
for 变量 in 值1 值2 值3...
do
程序/代码
done
案例:打印命令行输入的参数[这里可以看出$*和$@的区别]
#!/bin/bash
#$*把输入的参数当做一个整体,只输出一句
for i in "$*"
do
	echo "num is $i"
done
#$@来获取输入的参数,将它区分
for j in "$@"
do
	echo "num is $j"
done	
```

```
- 基本语法2
for((初始值;循环控制条件;变量变化))
do
程序
done
案例:从1加到100的值显示输出
SUM=0
for((i=1;i<=100;i++))
do
#业务代码
 SUM=$[$SUM + $i]
done
```

#### 4.while循环

```
- 基本语法1
while [ 条件判断式 ]
do
程序
done
案例:从命令行输入一个数n，统计从1+...+n的值是多少
#!/bin/bash  
SUM=0
i=0
while [ $i -le $1 ]
SUM=$[$SUM+$i]
#i自增
i=$[$i+1]
done
echo "执行结果=$SUM"
```
